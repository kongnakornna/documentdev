```java
/**
 * Service Layer - Business Logic
 * @Service: Marks as Spring-managed component
 */
@Service
@Transactional // Transaction management
public class UserService {
    
    private final UserRepository userRepository;
    private final PasswordEncoder passwordEncoder;
    
    // Constructor Injection
    @Autowired
    public UserService(
            UserRepository userRepository,
            PasswordEncoder passwordEncoder) {
        this.userRepository = userRepository;
        this.passwordEncoder = passwordEncoder;
    }
    
    /**
     * Business logic with transaction management
     * @Transactional ensures ACID properties
     */
    public UserDTO createUser(CreateUserRequest request) {
        // 1. Validate business rules
        validateUserData(request);
        
        // 2. Hash password
        String hashedPassword = passwordEncoder.encode(
            request.getPassword()
        );
        
        // 3. Create entity
        User user = User.builder()
            .username(request.getUsername())
            .email(request.getEmail())
            .passwordHash(hashedPassword)
            .build();
        
        // 4. Save to database (via Repository)
        User savedUser = userRepository.save(user);
        
        // 5. Convert to DTO and return
        return mapToDTO(savedUser);
    }
    
    public UserDTO findById(Long id) {
        User user = userRepository.findById(id)
            .orElseThrow(() -> 
                new UserNotFoundException("User not found: " + id)
            );
        
        return mapToDTO(user);
    }
}
```

### Service Layer with Redis Cache

```java
/**
 * User Service - Business logic for user management
 * Implements caching with Redis for performance optimization
 */
@Service
@Slf4j
public class UserService {
    
    @Autowired
    private UserRepository userRepository;
    
    @Autowired
    private RedisTemplate<String, User> redisTemplate;
    
    @Autowired
    private KafkaTemplate<String, UserEvent> kafkaTemplate;
    
    private static final String USER_CACHE_KEY = "user:";
    
    /**
     * Create new user with validation and event publishing
     * @param request User creation request DTO
     * @return Created user response
     */
    @Transactional
    public UserResponseDTO createUser(CreateUserRequest request) {
        // Validate username uniqueness
        if (userRepository.existsByUsername(request.getUsername())) {
            throw new DuplicateUserException("Username already exists");
        }
        
        // Hash password using BCrypt
        String hashedPassword = passwordEncoder.encode(request.getPassword());
        
        User user = User.builder()
            .username(request.getUsername())
            .email(request.getEmail())
            .passwordHash(hashedPassword)
            .role(UserRole.USER)
            .status(UserStatus.ACTIVE)
            .build();
        
        User savedUser = userRepository.save(user);
        
        // Cache user in Redis with TTL 1 hour
        redisTemplate.opsForValue().set(
            USER_CACHE_KEY + savedUser.getId(), 
            savedUser, 
            1, 
            TimeUnit.HOURS
        );
        
        // Publish user created event to Kafka
        publishUserEvent(savedUser, "USER_CREATED");
        
        return mapToResponseDTO(savedUser);
    }
    
    /**
     * Get user by ID with Redis caching
     * Cache-aside pattern implementation
     */
    @Cacheable(value = "users", key = "#userId")
    public UserResponseDTO getUserById(Long userId) {
        // Check Redis cache first
        User cachedUser = redisTemplate.opsForValue()
            .get(USER_CACHE_KEY + userId);
        
        if (cachedUser != null) {
            log.info("User found in cache: {}", userId);
            return mapToResponseDTO(cachedUser);
        }
        
        // If not in cache, fetch from database
        User user = userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("User not found"));
        
        // Store in cache
        redisTemplate.opsForValue().set(
            USER_CACHE_KEY + userId, 
            user, 
            1, 
            TimeUnit.HOURS
        );
        
        return mapToResponseDTO(user);
    }
}
```

```java
/**
 * Inventory Event Consumer
 * Listens to product and order events from Kafka
 */
@Component
@Slf4j
public class InventoryEventConsumer {
    
    @Autowired
    private InventoryService inventoryService;
    
    /**
     * Handle order created event - Reserve inventory
     */
    @KafkaListener(
        topics = "order-events",
        groupId = "inventory-service-group"
    )
    public void handleOrderEvent(OrderEvent event) {
        log.info("Received order event: {}", event);
        
        if (event.getEventType().equals("ORDER_CREATED")) {
            try {
                inventoryService.reserveInventory(
                    event.getOrderId(),
                    event.getOrderItems()
                );
                
                // Publish inventory reserved event
                publishInventoryEvent(event.getOrderId(), "INVENTORY_RESERVED");
                
            } catch (InsufficientStockException e) {
                log.error("Insufficient stock for order: {}", event.getOrderId());
                publishInventoryEvent(event.getOrderId(), "INVENTORY_INSUFFICIENT");
            }
        }
    }
    
    /**
     * Handle order cancelled event - Release inventory
     */
    @KafkaListener(
        topics = "order-events",
        groupId = "inventory-service-group"
    )
    public void handleOrderCancellation(OrderEvent event) {
        if (event.getEventType().equals("ORDER_CANCELLED")) {
            inventoryService.releaseInventory(
                event.getOrderId(),
                event.getOrderItems()
            );
        }
    }
}
```

```java
/**
 * JWT Utility Class - Token generation and validation
 */
@Component
public class JwtUtil {
    
    @Value("${jwt.secret}")
    private String jwtSecret;
    
    @Value("${jwt.expiration}")
    private Long jwtExpiration; // 24 hours
    
    /**
     * Generate JWT token with user details and roles
     */
    public String generateToken(UserDetails userDetails) {
        Map<String, Object> claims = new HashMap<>();
        claims.put("roles", userDetails.getAuthorities()
            .stream()
            .map(GrantedAuthority::getAuthority)
            .collect(Collectors.toList()));
        
        return Jwts.builder()
            .setClaims(claims)
            .setSubject(userDetails.getUsername())
            .setIssuedAt(new Date())
            .setExpiration(new Date(System.currentTimeMillis() + jwtExpiration))
            .signWith(SignatureAlgorithm.HS512, jwtSecret)
            .compact();
    }
    
    /**
     * Validate JWT token
     */
    public boolean validateToken(String token) {
        try {
            Jwts.parser().setSigningKey(jwtSecret).parseClaimsJws(token);
            return true;
        } catch (JwtException | IllegalArgumentException e) {
            log.error("Invalid JWT token: {}", e.getMessage());
            return false;
        }
    }
}
```

```groovy
pipeline {

                }
            }

                withSonarQubeEnv('SonarQube') {

                    docker-compose build
                '''
            }

                    docker-compose push
                '''
            }
        }
        
        stage('Deploy to AWS EC2') {
            steps {
                sh '''
                    ssh -i key.pem ec2-user@${EC2_HOST} \
                    "cd /app && docker-compose pull && docker-compose up -d"
                '''
            }
        }
    }
    
    post {
        success {
            slackSend color: 'good', 
                     message: "Pipeline succeeded: ${env.JOB_NAME} ${env.BUILD_NUMBER}"
        }
        failure {
            slackSend color: 'danger',

        }
    }
```

```java
/**
 * Standard API Response wrapper
 * Provides consistent response format across all services
 */
@Data
@Builder
public class ApiResponse<T> {
    
    private boolean success;
    private String message;
    private T data;
    private LocalDateTime timestamp;
    private Map<String, Object> metadata;
    
    public static <T> ApiResponse<T> success(T data) {
        return ApiResponse.<T>builder()
            .success(true)
            .data(data)
            .timestamp(LocalDateTime.now())
            .build();
    }
    
    public static <T> ApiResponse<T> success(T data, String message) {
        return ApiResponse.<T>builder()
            .success(true)
            .message(message)
            .data(data)
            .timestamp(LocalDateTime.now())
            .build();
    }
    
    public static <T> ApiResponse<T> error(String message) {
        return ApiResponse.<T>builder()
            .success(false)
            .message(message)
            .timestamp(LocalDateTime.now())
            .build();
    }
}
```
